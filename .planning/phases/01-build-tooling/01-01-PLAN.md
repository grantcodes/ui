    1. Install the Lit SSR DOM shim which provides a Node.js customization hook for CSS import attributes:
       ```
       pnpm add -D @lit-labs/ssr-dom-shim --filter @grantcodes/ui
       ```
       This package provides `@lit-labs/ssr-dom-shim/register-css-hook.js` which registers a Node.js module loader hook that intercepts `.css` file imports with `{ type: 'css' }` and returns a CSSStyleSheet shim object. The hook must be registered via `--import` before any component code loads.

    2. Update the `test:unit` script in `packages/ui/package.json` to register the CSS hook:
       Change from:
       ```
       "test:unit": "NODE_ENV=test node --test --test-concurrency=1 src/**/*.test.js"
       ```
       To:
       ```
       "test:unit": "NODE_ENV=test node --import @lit-labs/ssr-dom-shim/register-css-hook.js --test --test-concurrency=1 src/**/*.test.js"
       ```
       The `--import` flag must come before `--test` to ensure the CSS hook is registered before any test files import components. This is critical — if the hook registers after a component file tries to `import ... with { type: 'css' }`, it will fail.

    3. Create the test file `packages/ui/src/components/_css-test/css-test.test.js` following the existing test patterns (see button.test.js):
       ```javascript
       import { describe, it, afterEach } from "node:test";
       import { strict as assert } from "node:assert";
       import { fixture, cleanup } from "../../test-utils/index.js";
       import "./css-test.js";

       describe("CSS Import Attributes Test Fixture", () => {
         let element;

         afterEach(() => {
           cleanup(element);
         });

         it("should render with CSS import attributes", async () => {
           element = await fixture("grantcodes-css-test");
           assert.ok(element, "Element should be created");
           assert.ok(element.shadowRoot, "Element should have shadow root");
         });

         it("should have styles from CSS file", async () => {
           element = await fixture("grantcodes-css-test");
           // Verify the component class has styles defined (CSSStyleSheet from import)
           const styles = element.constructor.styles;
           assert.ok(styles, "Component should have static styles");
           assert.ok(styles.length > 0, "Styles array should not be empty");
         });

         it("should render default label", async () => {
           element = await fixture("grantcodes-css-test");
           const label = element.shadowRoot.querySelector(".css-test__label");
           assert.ok(label, "Label element should exist");
         });

         it("should accept custom label property", async () => {
           element = await fixture("grantcodes-css-test", {
             label: "Test Label",
           });
           assert.strictEqual(element.label, "Test Label", "Label property should be set");
         });
       });
       ```

    4. Run the tests to verify:
       ```
       pnpm --filter @grantcodes/ui test:unit
       ```
       All existing tests should still pass AND the new _css-test tests should pass. If existing tests break due to the `--import` flag, investigate — the CSS hook should be a no-op for files that don't use CSS import attributes.

    5. If the `@lit-labs/ssr-dom-shim/register-css-hook.js` approach doesn't work with the happy-dom setup (possible conflict between happy-dom's CSSStyleSheet and the shim), fall back to writing a minimal custom Node.js loader. Create `packages/ui/css-import-loader.js`:
       ```javascript
       import { readFile } from "node:fs/promises";

       export async function load(url, context, nextLoad) {
         if (context.importAttributes?.type === "css") {
           const source = await readFile(new URL(url), "utf-8");
           return {
             format: "module",
             source: `
               const sheet = new CSSStyleSheet();
               sheet.replaceSync(${JSON.stringify(source)});
               export default sheet;
             `,
             shortCircuit: true,
           };
         }
         return nextLoad(url, context);
       }
       ```
       Then update test:unit to use `--import ./css-import-loader.js` instead.
       Only use this fallback if the ssr-dom-shim approach fails. The ssr-dom-shim is the Lit team's official solution.
